``` r
instacart |> 
  filter(
    aisle == "baking ingredients" |
    aisle == "dog food care" | 
    aisle == "packaged vegetables fruits") |> 
  group_by(aisle, product_name) |> 
  summarize(n_orders = n()) |> 
  mutate(
    n_order_rank = min_rank(desc(n_orders))
  ) |> 
  filter(n_order_rank <= 3) |> 
  select(product_name, aisle, n_orders) |> 
  knitr::kable()
```
